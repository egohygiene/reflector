% Canonical Paper Title
%
% Single source of truth for the paper title.
% Edit \papertitlemain and \papertitlesubtitle to match your
% publication. \papertitlefull is assembled automatically.
%
% Load this file before the publication style so that title
% macros are available for hyperref, headers, and other
% style-level configuration.
%
% Macros defined here:
%   \papertitlemain      — primary title token
%   \papertitlesubtitle  — subtitle string (leave empty if none)
%   \papertitlefull      — assembled canonical title
% ============================================================

  An Example Research Publication%
}

\newcommand{\papertitlesubtitle}{%
  A Generic Template for Research Papers%
}

\newcommand{\papertitlefull}{%
  \papertitlemain: \papertitlesubtitle%
}
